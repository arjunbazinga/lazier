@lazier
def run(n, alpha, p, iterations = 10000, returns=False):
    z = np.random.randn(n,1)
    activation = []
    ys = []
    costs = []
    for i in range(iterations):
        y = gen_y(p)
        ys.append(y)

        a =  softmax(z)
        activation.append(a)

        c = cost(y , a)

        costs.append(c)

        dc_dz = a - y

        z = z - (alpha * dc_dz)
        
    m = get_measured_prob(ys)

    for i in range(n):
        plt.plot([j[i] for j in m], label = f"measured prob {i}")
        plt.plot([j[i] for j in activation], label = f"predicted prob {i}")
    plt.show()
    
    if returns:
        return  z,activation,ys,costs
